处理模数不互质的情况
```cpp
ll ExCRT(vector<ll>&r,vector<ll>&a) {
	ll R = r[1], A = a[1];
	auto mix = [&](ll aa, ll rr) {
		ll x, y;
		liEu(R, rr, aa - A, x, y);
		A += R * x;
		R = lcm(R, rr);
		A %= R;
		if (A < 0)A += R;
		};
	for (int i = 2; i < r.size(); i++)
		mix(a[i], r[i]);
	ll ans = A % R;
	if (ans < 0)ans += R;
	return ans;
}
```
