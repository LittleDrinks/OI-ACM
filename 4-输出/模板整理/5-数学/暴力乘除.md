using poly = vector<ll>;
poly operator*(const poly &a, const poly &b) {
    int n = a.size();
    int m = b.size();
    poly c(n + m - 1);
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < m; ++j) {
            c[i + j] = add(c[i + j], mul(a[i], b[j]));
        }
    }
    return c;
}
poly operator/(poly a, poly b) {
    int n = a.size();
    int m = b.size();
    if (b[1] == 0) {
        return a;
    }

    poly c(n - 1);
    ll base = qpow(b[1]);
    for (int i = n - 1; i; --i) {
        ll d = mul(a[i], base);
        c[i - 1] = d;
        a[i - 1] = del(a[i - 1], mul(d, b[0]));
    }
    return c;
}
