树上任意两节点之间最长的简单路径
树形 DP 求法
```cpp
int d[N], mxd;
void dfs(int u, int fa)
{
	for (auto v: G[u]) {
		if (v != fa) {
			dfs(v, u);
			mxd = max(mxd, d[u]+d[v]+1);
			d[u] = max(d[u], d[v]+1);
		}
	}
}
```
两次 DFS 求法，可以把整条直径提取出来，然后在其上找到树的中心
```cpp
void work() {
	int mxd=0, V=0;
	vector<int> f(n+1);
	function<void(int,int,int)> dfs = [&](int u, int fa, int d){
	    f[u] = fa;
	    if (d > mxd) {
	        mxd = d;
	        V = u;
	    }
	    for (auto v: G[u]) {
	        if (v != fa) {
	            dfs(v, u, d+1);
	        }
	    }
	};
	
	// 第一次 dfs，找端点
	mxd = 0;
	dfs(1, 0, 1);
	
	// 第二次 dfs，走直径
	mxd = 0;
	int s = V;
	dfs(V, 0, 1);
	int t = V;
	
	// 把直径存入 vector
	vector<int> R{t};
	while (t != s) {
	    t = f[t];
	    R.push_back(t);
	}
}
```
