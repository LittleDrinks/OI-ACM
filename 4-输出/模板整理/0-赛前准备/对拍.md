.sh 脚本
```cpp
#!/bin/bash
P=$1
g++ -std=gnu++20 -O2 "${P}.cpp" -o "${P}"
g++ -std=gnu++20 -O2 "${P}_bf.cpp" -o "${P}_bf"
g++ -std=gnu++20 -O2 "${P}_gen.cpp" -o "${P}_gen"
while true; do
    ./"${P}_gen" > ".in"
    ./"${P}" < ".in" > ".out"
    ./"${P}_bf" < ".in" > ".ans"
    if diff ".out" ".ans" > /dev/null; then
        echo "AC"
    else
        echo "WA"
        break
    fi
done
```
