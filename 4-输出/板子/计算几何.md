typedef long long ll;
typedef double db;  // 视情况改为 long double
const db eps=1e-8;  // 极端数据下，放大 eps 可能有奇效
const db PI=acos(-1.0);
int sgn(db x) { return fabs(x)<eps? 0: ( x>0? 1: -1 ); }
int sgn(ll x) { return x==0? 0: ( x>0? 1: -1 ); }
