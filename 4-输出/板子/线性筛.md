vector <int> getPrime(int n)
{
	vector <bool> vis(n+1);
	vector <int> prime;
	vis[1] = 1;
	for (int i = 2; i <= n; ++i) {
		if (!vis[i]) { prime.push_back(i); }
		for (auto p: prime) {
			if (i*p > n) { break; }
			vis[i*p] = 1;
			if (i%p == 0) { break; }  // 线性筛关键优化
		}
	}
	return prime;
}
