const int MOD=998244353;
ll add(ll x, ll y) { return (x+y)%MOD; }
ll del(ll x, ll y) { return add(x, MOD-y); }
ll mul(ll x, ll y) { return (x*y)%MOD; }
ll qpow(ll a, ll b = MOD - 2) {
	ll res = 1;
	for (; b; b >>= 1) {
		if (b & 1) { res = mul(res, a); }
		a = mul(a, a);
	}
	return res;
}
