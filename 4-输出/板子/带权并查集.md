```cpp
struct DSU {
    int n;
    vector<int> f, siz, pre;
    DSU(int n)
        : n(n), f(vector<int>(n + 1)), siz(vector<int>(n + 1, 1)),
          pre(vector<int>(n + 1)) {
        iota(f.begin(), f.end(), 0);
    }
    int find(int x) {
        if (f[x] == x) {
            return x;
        }
        int fa = find(f[x]);
        pre[x] += pre[f[x]];
        return f[x] = fa;
    }
    bool same(int x, int y) {
        return find(x) == find(y);
    }
    void merge(int x, int y) {
        x = find(x);
        y = find(y);
        f[x] = y;
        pre[x] += siz[y];
        siz[y] += siz[x];
        siz[x] = 0;
    }
    int query(int x, int y) {
        return abs(pre[x] - pre[y]) - 1;
    }
};
```
