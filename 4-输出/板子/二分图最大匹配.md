```cpp
#include <bits/stdc++.h>

using namespace std;

const int N=505, M=505, E=505;
int n, m, e, match[N], ans;
bool vis[N];
vector <int> G[N];

bool dfs(int u)
{
    for (int v: G[u]) {
        if (!vis[v]) {
            vis[v] = true;
            if (!match[v] || dfs(match[v])) {
                match[v] = u; return true;
            }
        }
    }
    return false;
}

int main()
{
    cin >> n >> m >> e;
    for (int i = 1; i <= e; ++i) {
        int u, v;
        cin >> u >> v;
        G[u].push_back(v);
    }
    for (int i = 1; i <= n; ++i) {
        memset(vis, 0, sizeof(vis));
        ans += dfs(i);
    }
    cout << ans << endl;
}
```
