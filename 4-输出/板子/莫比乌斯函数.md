vector <int> get_Mu(int n)
{
	vector<int> mu(n + 1);
	vector <bool> vis(n + 1);
	vector <int> prime;
	vis[1] = 1;
	for (int i = 2; i <= n; ++i) {
		if (!vis[i]) { prime.push_back(i); mu[i] = -1; }
		for (auto p : prime) {
			if (i * p > n) { break; }
			vis[i * p] = 1;
			if (i % p == 0) { mu[i * p] == 0; break; }  
			mu[i * p] = -mu[i];
		}
	}
	return mu;
}
