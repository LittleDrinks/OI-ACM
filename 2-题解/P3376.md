## [P3376 【模板】网络最大流](https://www.luogu.com.cn/problem/P3376)

最大流模板题。

[***AC 代码***](https://www.luogu.com.cn/record/208714784)

```cpp
#include <bits/stdc++.h>

using namespace std;
typedef long long ll;
const int N=205;

struct Flow {
    const ll INF = 9e18; 
    vector<tuple<int,ll,int>> G[N];  // 终点，流量，反边
    int dis[N], now[N];  // now[] 为当前弧
    void add_edge(int u, int v, ll w) {
        G[u].push_back( { v, w, G[v].size() } );
        G[v].push_back( { u, 0, G[u].size()-1 } );
    }
    void bfs(int st) {  // 求层次图
        queue<int> q;
        memset(dis, 0, sizeof(dis));
        q.push(st); dis[st] = 1;
        while (!q.empty()) {
            int u = q.front(); q.pop();
            for (auto& [v, w, inv]: G[u]) {
                if (w != 0 && dis[v] == 0) {
                    dis[v] = dis[u] + 1;
                    q.push(v);
                }
            }
        }
    }
    ll dfs(int u, int t, ll flow) {
        if (u == t) { return flow; }
        for (int &i=now[u]; i < (int)G[u].size(); ++i) {
            auto& [v, w, inv] = G[u][i];
            if (w != 0 && dis[v] > dis[u]) {
                ll d = dfs(v, t, min(flow, w));
                if (d > 0) {
                    w -= d;
                    get<1>(G[v][inv]) += d;
                    return d;
                }
            }
        }
        return 0;
    }
    ll dicnic(int st, int ed) {
        for (ll flow=0, res;;) {
            bfs(st);
            if (dis[ed] == 0) { return flow; }
            memset(now, 0, sizeof(now));
            while ((res=dfs(st,ed,INF)) > 0) {
                flow += res;
            }
        }
    }
};

int main()
{
	cin.tie(nullptr)->sync_with_stdio(false);
    int n, m, s, t;
    cin >> n >> m >> s >> t;
    Flow mxf;
    for (int i = 1; i <= m; ++i) {
        int u, v; ll w;
        cin >> u >> v >> w;
        mxf.add_edge(u, v, w);
    }
    cout << mxf.dicnic(s, t) << "\n";
}
```
