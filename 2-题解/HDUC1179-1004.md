## [HDUC1179-1004. 最糖的题目](https://acm.hdu.edu.cn/contest/problem?cid=1179&pid=1004)

结论题。
先特判构成 $a,b$ 的可重集不一样的情况，以及 $k=1$ 和 $k=n$ 的情况。
$k=n$ 的情况可以使用 KMP 在 $O(n)$ 的时间内完成。
考虑排列的情况。一次 shift 操作相当于把末尾的数冒泡到开头，相当于 $k-1$ 次交换操作，每次交换操作都会改变一次逆序数的奇偶性。因此如果 $k$ 为奇数一定可行，$k$ 为偶数时逆序数奇偶性相同则一定可行。
考虑非排列的情况，实际上就是给每个数打一个标签让他变成排列，然后值相同的位置标签可以互换。可以发现，如果存在两个值相同，则一定可以通过互换他们两个的标签使逆序数奇偶性改变。此时一定可行。

[***AC 代码***](https://acm.hdu.edu.cn/contest/view-code?cid=1179&rid=13216)

```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using db = long double;
using i128 = __int128;

const int N=100000;
int n, k, t[N+5];
int lowbit(int x) { return x&-x; }
void modify(int x, int d) {
    for (; x <= n; x += lowbit(x)) { t[x] += d; }
}
int query(int x) {
    int res = 0;
    for (; x; x -= lowbit(x)) { res += t[x]; }
    return res;
}
int inverse_num(auto& vec) {
    int ans = 0;
    memset(t, 0, sizeof(t));
    for (int i = 0; i < n; ++i) {
        ans = (ans + i - query(vec[i])) % 2;
        modify(vec[i], 1);
    }
    return ans;
};

int fail[N<<2], s[N<<2], a[N+5], b[N+5];
void prefixFunction()
{   // fail : 1-index
    for (int i = 1, j = 0; i <= 3*n; ++i) {
        while (j && s[i] != s[j]) { j = fail[j]; }
        j += (s[i] == s[j]);
        fail[i + 1] = j;
    }
}
bool kmp()
{
    for (int i = 0; i < n; ++i) { s[i] = b[i]; }
    s[n] = -1;
    for (int i = 0; i < n; ++i) { s[n+i+1] = a[i]; }
    for (int i = 0; i < n; ++i) { s[2*n+i+1] = a[i]; }
    prefixFunction();
	for (int i = n + 2; i < 3 * n + 1; ++i) {
	    if (fail[i] == n) {
            return true;
	    }
	}
    return false;
}

void solve()
{
    cin >> n >> k;

    // 读入 a
    bool pa = true;
    unordered_map<int,int> ca, cb;
    vector<int> v;
    for (int i = 0; i < n; ++i) {
        auto &x = a[i];
        cin >> x;
        ++ca[x];
        v.push_back(x);
        if (ca[x] > 1) { pa = false; }
    }

    // 对 a 离散化
    sort(v.begin(), v.end());
    v.erase(unique(v.begin(), v.end()), v.end());
    for (int i = 0; i < n; ++i) {
        auto &x = a[i];
        x = lower_bound(v.begin(), v.end(), x) - v.begin() + 1;
    }
    
    // 读入 b
    for (int i = 0; i < n; ++i) {
        auto &x = b[i];
        cin >> x;
        ++cb[x];
    }
    // a、b 数字是否相同
    for (int i = 0; i < n; ++i) {
        auto &x = b[i];
        if (ca[x] != cb[x]) { cout << "NO\n"; return; }
    }
    // 对 b 离散化
    for (int i = 0; i < n; ++i) {
        auto &x = b[i];
        x = lower_bound(v.begin(), v.end(), x) - v.begin() + 1;
    }

    if (k == 1) {
        for (int i = 1; i <= n; ++i) {
            if (a[i] != b[i]) { cout << "NO\n"; return; }
        }
        cout << "YES\n";
    } else if (n == k) {
        bool valid = kmp();
        cout << (valid ? "YES" : "NO") << "\n";
    } else if (pa) {  // 是排列
        bool valid = true;
        if (k % 2 == 1) {
            valid &= inverse_num(a) == inverse_num(b);          
        }
        cout << (valid ? "YES" : "NO") << "\n";
    } else {
        cout << "YES\n";
    }
}

int main()
{
    ios::sync_with_stdio(0); cin.tie(0); cout.tie(0); 
    int t = 1;
    cin >> t;
    while (t--) { solve(); }
}
```
