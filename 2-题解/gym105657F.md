## [gym105657F. Fuzzy Ranking](https://codeforces.com/gym/105657/problem/F)

由 $a_{i,j-1}$ 向 $a_{i,j}$ 建边，求 SCC。
记询问每个 SCC 包含区间内的节点的个数为 $c_i$，每次询问实际上是在求 $$\sum_i{c_i\choose 2}$$
区间上每个 SCC 的编号都是连续的，可以用前缀和优化计算过程。
实现时分为两类，第一种是整个区间都在一个 SCC 内，直接计算 $r-l+1\choose2$ 即可。否则去除前后两个不完整 SCC 后，中间使用前缀和计算。

[***AC 代码***](https://codeforces.com/gym/105657/submission/314867201)

```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef pair<int,ll> pii;
struct SCC {
    int n, totdfn=0, colcnt=0;
    vector<int> stk;
    vector<int> dfn, low, col;
    vector<vector<int>> G;
    SCC(int n): n(n) {
        G.assign(n+1, vector<int>());
        dfn.assign(n+1, 0);
        low.assign(n+1, 0);
        col.assign(n+1, 0);
    }
    void add_edge(int u, int v) {
        G[u].push_back(v);
    }
    void dfs(int u) {
        stk.push_back(u);
        dfn[u] = low[u] = ++totdfn;
        for (auto v: G[u]) {
            if (!dfn[v]) {
                dfs(v);
                low[u] = min(low[u], low[v]);
            } else if (!col[v]) {
                low[u] = min(low[u], dfn[v]);
            }
        }
        if (low[u] == dfn[u]) {
            col[u] = ++colcnt;
            int v;
            do {
                v = stk.back(); stk.pop_back();
                col[v] = colcnt;
            } while (v != u);
        }
    }
    vector<int> work() {
        for (int i = 1; i <= n; ++i) {
            if (!dfn[i]) { dfs(i); }
        }
        return col;
    }
};
void solve()
{
    int n, k, q;
    cin >> n >> k >> q;

    // 求 SCC
    SCC scc(n);
    vector<vector<int>> a(k+1, vector<int>(n+1));
    for (int i = 1; i <= k; ++i) {
        for (int j = 1; j <= n; ++j) { cin >> a[i][j]; }
        for (int j = 2; j <= n; ++j) {
            scc.add_edge(a[i][j-1], a[i][j]);
        }
    }
    vector<int> col = scc.work();

    // 求 每个列表的前缀和
    vector<set<pii>> pre(k+1);
    for (int i = 1; i <= k; ++i) {
        int cnt = 0;
        ll sum = 0;
        for (int j = 1; j <= n; ++j) {
            if (col[a[i][j]] != col[a[i][j-1]]) {
                sum += 1LL*cnt*(cnt-1)/2;
                pre[i].insert( { j-1, sum } );
                cnt = 1;
            } else { ++cnt; }
        }
        sum += 1LL*cnt*(cnt-1)/2;
        pre[i].insert( { n, sum } );
    }

    // 分为两部分处理询问
    // 1. 前后两个断点
    // 2. 中间的连续段
    int v = 0;
    while (q--) {
        int id, l, r;
        cin >> id >> l >> r;
        id = (id + v) % k + 1;
        l = (l + v) % n + 1;
        r = (r + v) % n + 1;
        auto it1 = pre[id].upper_bound( { r, -1 } );
        --it1;
        if (it1->first < l) { v = 1LL*(r-l+1)*(r-l)/2; }
        else {
            auto it2 = pre[id].lower_bound( { l, -1 } );
            v = it1->second - it2->second;
            ll nr = r - it1->first;
            ll nl = it2->first - l + 1;
            v += nr*(nr-1)/2 + nl*(nl-1)/2;
        }
        cout << v << "\n";
    }
}
int main()
{
    cin.tie(nullptr)->sync_with_stdio(false);
    int t; cin >> t;
    while (t--) { solve(); }
}
```
