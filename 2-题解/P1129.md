## [P1129 [ZJOI2007] 矩阵游戏](https://www.luogu.com.cn/problem/P1129)



[***AC 代码***](https://www.luogu.com.cn/record/208870557)

```cpp
#include <bits/stdc++.h>

using namespace std;
const int N=405;
const int INF = 2e9;

struct Flow {
    vector<tuple<int,int,int>> G[N];
    int now[N], dis[N];
    void add_edge(int u, int v, int w) {
        G[u].push_back( { v, w, G[v].size() } );
        G[v].push_back( { u, 0, G[u].size()-1 } );
    }
    void bfs(int s) {
        queue<int> q;
        memset(dis, 0, sizeof(dis));
        q.push(s); dis[s] = 1;
        while (!q.empty()) {
            int u = q.front(); q.pop();
            for (auto& [v, w, inv]: G[u]) {
                if (w != 0 && dis[v] == 0) {
                    dis[v] = dis[u] + 1;
                    q.push(v);
                }
            }
        }
    }
    int dfs(int u, int t, int flow) {
        if (u == t) { return flow; }
        for (int &i = now[u]; i < (int)G[u].size(); ++i) {
            auto& [v, w, inv] = G[u][i];
            if (w != 0 && dis[v] > dis[u]) {
                int d = dfs(v, t, min(flow, w));
                if (d != 0) {
                    w -= d;
                    get<1>(G[v][inv]) += d;
                    return d;
                }
            }
        }
        return 0;
    }
    int dicnic(int s, int t) {
        for (int flow=0, res;;) {
            bfs(s);
            if (dis[t] == 0) { return flow; }
            memset(now, 0, sizeof(now));
            while ((res=dfs(s, t, INF)) > 0) {
                flow += res;
            }
        }
    }
};

void solve()
{
    Flow mxf;
    int n; cin >> n;
    int s=0, t=(n+1)<<1;
    for (int i = 1; i <= n; ++i) {
        mxf.add_edge(s, i<<1, 1);
        mxf.add_edge(i<<1|1, t, 1);
    }
    for (int i = 1; i <= n; ++i) {
        for (int j = 1; j <= n; ++j) {
            int x; cin >> x;
            if (x == 1) {
                mxf.add_edge(i<<1, j<<1|1, 1);
            }
        }
    }
    if (mxf.dicnic(s, t) == n) { cout << "Yes\n"; }
    else { cout << "No\n"; }
}

int main()
{
	cin.tie(nullptr)->sync_with_stdio(false);
	int t; cin >> t;
	while (t--) { solve(); }
}
```
