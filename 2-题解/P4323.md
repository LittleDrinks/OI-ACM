## [P4323 [JSOI2016] 独特的树叶](https://www.luogu.com.cn/problem/P4323)

采用 $f(u)=\sum f(v)\times P(siz_v)$ 进行哈希，其中 $P(k)$ 表示第 $k$ 个质数。
记 $g(u)$ 表示以 $u$ 为根时整棵树的哈希值，有换根转移式
$$
g(u) = (g(fa)-f(u)\times P(siz_u))\times P(n-siz_u) + f(u)
$$

[***AC 代码***](https://www.luogu.com.cn/record/224299215)

```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using db = long double;
using ull = unsigned long long;
using i128 = __int128;

vector<ll> sieve(int n)
{
	vector<bool> vis(n+1);
	vector<ll> prime;
	vis[1] = 1;
	for (int i = 2; i <= n; ++i) {
		if (!vis[i]) { prime.push_back(i); }
		for (auto p: prime) {
			if (i*p > n) { break; }
			vis[i*p] = 1;
			if (i%p == 0) { break; }  // 线性筛关键优化
		}
	}
	return prime;
}
vector<ll> P = sieve(2e6);

const int MOD = 998244353;
template<typename T> T add(T x, T y) { return (x + y) % MOD; }
template<typename T> T del(T x, T y) { return add(x, MOD - y); }
template<typename T> T mul(T x, T y) { return (x * y) % MOD; }

const int N=1e5+5;
int n, siz[N], deg[N];
vector<int> G[N];
ll f[N], g[N];

void dfs(int u, int fa)
{
    siz[u] = 1;
    f[u] = 1;
    for (auto v: G[u]) {
        if (v == fa) {
            continue; 
        }
        dfs(v, u);
        siz[u] += siz[v];
        f[u] = add(f[u], mul(f[v], P[siz[v]]));
    }
}

void change(int u, int fa)
{
    if (fa) {
        ll ffa = del(g[fa], mul<ll>(f[u], P[siz[u]]));
        g[u] = add(f[u], mul<ll>(ffa, P[n-siz[u]]));
    }
    for (auto v: G[u]) {
        if (v == fa) {
            continue;
        }
        change(v, u);
    }
}

void solve()
{
	cin >> n;
    for (int i = 1; i < n; ++i) {
        int u, v;
        cin >> u >> v;
        G[u].push_back(v);
        G[v].push_back(u);
    }
    dfs(1, 0);
    g[1] = f[1];
    change(1, 0);
    set<ll> tr;
    for (int i = 1; i <= n; ++i) {
        tr.insert(g[i]);
    }
    for (int i = 1; i <= n; ++i) {
        G[i].clear();
        f[i] = g[i] = 0;
    }

    ++n;
    for (int i = 1; i < n; ++i) {
        int u, v;
        cin >> u >> v;
        G[u].push_back(v);
        G[v].push_back(u);
        ++deg[u]; ++deg[v];
    }
    dfs(1, 0);
    g[1] = f[1];
    change(1, 0);

    for (int i = 1; i <= n; ++i) {
        if (deg[i] != 1) {
            continue;
        }
        ll gg = del<ll>(g[G[i][0]], P[1]);
        if (tr.count(gg)) {
            cout << i << "\n";
            return;
        }
    }
    assert(false);
}

int main()
{
	cin.tie(nullptr)->sync_with_stdio(false);
	int t = 1;
	// cin >> t;
	while (t--) { solve(); }
}
```
