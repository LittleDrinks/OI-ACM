## [gym104077L. Tree](https://codeforces.com/gym/104077/problem/L)

第一类集合等价于将树拆成若干条互不相交的链，链的数量等于叶子的数量。
每次可以将若干个叶子合并成第二类集合，此时叶子数量单调递减。
枚举第二类集合的个数，拓扑排序维护即可，时间复杂度 $O(n)$。

[***AC 代码***](https://codeforces.com/gym/104077/submission/343841945)

```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

void solve()
{
    int n;
    cin >> n;
    vector<int> fa(n + 1), deg(n + 1);
    for (int i = 2; i <= n; ++i) {
        cin >> fa[i];
        ++deg[fa[i]];
    }
    queue<int> q;
    for (int i = 1; i <= n; ++i) {
        if (deg[i] == 0) {
            q.push(i);
        }
    }
    int cnt = 0;
    int ans = q.size();
    while (!q.empty()) {
        queue<int> tmp;
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            if (--deg[fa[u]] == 0) {
                tmp.push(fa[u]);
            }
        }
        ++cnt;
        ans = min(ans, cnt + int(tmp.size()));
        if (tmp.size() == 1) {
            break;
        }
        while (!tmp.empty()) {
            q.push(tmp.front());
            tmp.pop();
        }
    }
    cout << ans << "\n";
}

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int t = 1;
    cin >> t;
    while (t--) {
        solve();
    }
}
```
